git init . - cria um repositório vazio na pasta selecionada
git status - mostra o status do git
git add {arquivo}  - aficionado arquivos para o commit
git commit -m "(add sua msg aqui)" - comita os arquivos adicionados e adiciona uma frase.
git config --global user.email "(add email)"  - adiciona email 
git config --global user.name "(add nome)" - adiciona nome
git config --global user.email  - verifica o que tem salvo
git config --global user.name  - verifica o que tem salvo
git log - mostra todos os commits
git log --oneline  - mostra de forma resumida
git diff <nome do arquivo> - diferença entre o estado atual e o ultimo commit
git reset <nome do arquivo> - remove do stage
Git checkout -b <nome da branch> - cria uma branch paralela - -b cria branch nova
git checkout <nome da branch> - muda de branch
git branch - mostra todas as branchs  
git merge <nome da branch> - traz para a branch atual as alterações feitas na branch descrita <>
